// Setiap content block punya: DATA + COMPONENT TYPE
interface ContentBlock {
  id: string
  type: 'text' | 'table' | 'accordion' | 'card' 
      | 'tabs' | 'flip-card' | 'callout' 
      | 'stepper' | 'carousel' | 'drag-drop'
      | 'fill-blank' | 'tooltip-text'
  
  // Data mentah (guru isi)
  data: Record<string, any>
  
  // Komponen render yang dipilih
  component: ComponentType
}

// Contoh: data tabel
{
  type: 'table',
  component: 'table',
  data: {
    headers: ['Sifat', 'Logam', 'Non-Logam'],
    rows: [
      ['Konduksi', 'Konduktor', 'Isolator'],
      ['Kilap', 'Mengkilap', 'Tumpul'],
    ]
  }
}

// Contoh: data accordion
{
  type: 'text',
  component: 'accordion',
  data: {
    items: [
      { title: 'Apa itu Logam?', content: 'Logam adalah...' },
      { title: 'Sifat Logam', content: 'Logam memiliki...' },
    ]
  }
}

Kenapa Wajib

8	Drag & Drop Match	Kinesthetic learning	Cocokkan istilah↔definisi

Pendekatan 1: Headless Component Library (RECOMMENDED)
text

┌──────────────────────────────────────────────┐
│  JANGAN tulis UI dari nol                    │
│  GUNAKAN headless component + style sendiri  │

→ Kamu kontrol 100% tampilan
→ Tapi logic (accordion buka-tutup, drag, dll) sudah jadi
